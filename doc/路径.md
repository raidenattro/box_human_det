import math
import csv
from collections import deque

class ActionFeatureExtractor:
    def __init__(self, csv_filename="action_dataset.csv"):
        self.history = {}
        self.csv_filename = csv_filename
        self.csv_file = open(self.csv_filename, mode='a', newline='', encoding='utf-8')
        self.csv_writer = csv.writer(self.csv_file)
        
        # 写入表头 (如果文件是空的)
        if os.path.getsize(self.csv_filename) == 0:
            headers = [
                "frame_idx", "person_id", "hand_side", 
                "norm_x", "norm_y", "score",          # 归一化位置
                "v_x", "v_y", "a_x", "a_y",           # 速度与加速度
                "is_in_box", "dist_to_box_center", "box_id" # 环境特征
            ]
            self.csv_writer.writerow(headers)

    def extract_and_save(self, frame_idx, person_id, kpts, scores, boxes):
        """
        kpts: [17, 2] 的坐标数组
        scores: [17] 的置信度数组
        boxes: 当前系统里的多边形列表
        """
        # 1. 找原点（脖子：左右肩 5 和 6 的中点）和 标尺（肩宽）
        if scores[5] < 0.2 or scores[6] < 0.2:
            return  # 肩膀被遮挡，无法建立坐标系，跳过

        neck_x = (kpts[5][0] + kpts[6][0]) / 2.0
        neck_y = (kpts[5][1] + kpts[6][1]) / 2.0
        shoulder_width = math.hypot(kpts[5][0] - kpts[6][0], kpts[5][1] - kpts[6][1])
        if shoulder_width < 10: 
            shoulder_width = 10.0 # 防止除以 0

        # 2. 提取左右手特征 (9: 左手腕, 10: 右手腕)
        for hand_side, wrist_idx in [("left", 9), ("right", 10)]:
            score = scores[wrist_idx]
            if score < 0.15: continue # 手完全看不见，跳过
            
            wrist_x, wrist_y = kpts[wrist_idx]
            
            # 【A. 空间归一化】
            norm_x = (wrist_x - neck_x) / shoulder_width
            norm_y = (wrist_y - neck_y) / shoulder_width
            
            # 【B. 运动学计算 (使用队列记录历史)】
            hand_key = f"{person_id}_{hand_side}"
            if hand_key not in self.history:
                self.history[hand_key] = deque(maxlen=3)
            
            history = self.history[hand_key]
            history.append((norm_x, norm_y))
            
            v_x, v_y, a_x, a_y = 0.0, 0.0, 0.0, 0.0
            if len(history) >= 2:
                v_x = history[-1][0] - history[-2][0]
                v_y = history[-1][1] - history[-2][1]
            if len(history) >= 3:
                prev_v_x = history[-2][0] - history[-3][0]
                prev_v_y = history[-2][1] - history[-3][1]
                a_x = v_x - prev_v_x
                a_y = v_y - prev_v_y

            # 【C. 环境交互特征】寻找手在哪个框里，或者离哪个框最近
            is_in_box = 0
            target_box_id = -1
            min_dist = float('inf')
            
            for box in boxes:
                contour = box['orig_contour']
                # 计算多边形中心点
                M_moments = cv2.moments(contour)
                if M_moments["m00"] != 0:
                    cx = int(M_moments["m10"] / M_moments["m00"])
                    cy = int(M_moments["m01"] / M_moments["m00"])
                else:
                    cx, cy = contour[0][0][0], contour[0][0][1]

                # 相对距离也除以肩宽进行归一化
                dist_to_center = math.hypot(wrist_x - cx, wrist_y - cy) / shoulder_width
                
                # 判断是否在框内
                if cv2.pointPolygonTest(contour, (float(wrist_x), float(wrist_y)), False) >= 0:
                    is_in_box = 1
                    target_box_id = box['box_id']
                    min_dist = dist_to_center
                    break # 已经在框里了，直接锁定这个框
                
                # 如果不在框内，记录离得最近的框
                if dist_to_center < min_dist:
                    min_dist = dist_to_center
                    target_box_id = box['box_id']

            # 3. 保存这一帧的特征向量到 CSV
            row = [
                frame_idx, person_id, hand_side,
                round(norm_x, 4), round(norm_y, 4), round(score, 4),
                round(v_x, 4), round(v_y, 4), round(a_x, 4), round(a_y, 4),
                is_in_box, round(min_dist, 4), target_box_id
            ]
            self.csv_writer.writerow(row)

# 全局实例化这个特征提取器
feature_extractor = ActionFeatureExtractor()

### B. 新增一个轻量手部轨迹关联器（不依赖 ReID）

```python
@dataclass
class TrackState:
    norm_x: float
    norm_y: float
    ts_sec: float


class WristTrackAssigner:
    """在单摄像头场景下，用近邻匹配给手部生成稳定 track_id。"""

    def __init__(self, max_match_dist=0.35, stale_sec=1.0):
        self.max_match_dist = max_match_dist
        self.stale_sec = stale_sec
        self.next_id = 1
        self.tracks = {}  # key: (hand_side, track_id) -> TrackState

    def _cleanup(self, now_ts: float):
        dead_keys = []
        for k, st in self.tracks.items():
            if now_ts - st.ts_sec > self.stale_sec:
                dead_keys.append(k)
        for k in dead_keys:
            self.tracks.pop(k, None)

    def assign(self, hand_side: str, norm_x: float, norm_y: float, now_ts: float) -> int:
        self._cleanup(now_ts)

        best_key = None
        best_dist = 1e9
        for (side, tid), st in self.tracks.items():
            if side != hand_side:
                continue
            d = math.hypot(norm_x - st.norm_x, norm_y - st.norm_y)
            if d < best_dist:
                best_dist = d
                best_key = (side, tid)

        if best_key is None or best_dist > self.max_match_dist:
            tid = self.next_id
            self.next_id += 1
            self.tracks[(hand_side, tid)] = TrackState(norm_x=norm_x, norm_y=norm_y, ts_sec=now_ts)
            return tid

        _, tid = best_key
        self.tracks[(hand_side, tid)] = TrackState(norm_x=norm_x, norm_y=norm_y, ts_sec=now_ts)
        return tid
```

class ActionFeatureExtractorV2:
    def __init__(self, csv_filename="action_dataset.csv", auto_flush_every=200):
        self.history = {}
        self.csv_filename = csv_filename
        self.auto_flush_every = auto_flush_every
        self.write_count = 0

        os.makedirs(os.path.dirname(csv_filename) or ".", exist_ok=True)
        self.csv_file = open(self.csv_filename, mode='a', newline='', encoding='utf-8')
        self.csv_writer = csv.writer(self.csv_file)

        if os.path.getsize(self.csv_filename) == 0:
            headers = [
                "session_id", "video_id", "ts_sec", "frame_idx",
                "track_id", "person_id_raw", "hand_side",
                "norm_x", "norm_y", "score",
                "v_x", "v_y", "a_x", "a_y",
                "is_in_box", "dist_to_box_center", "box_id"
            ]
            self.csv_writer.writerow(headers)

    def _save_row(self, row):
        self.csv_writer.writerow(row)
        self.write_count += 1
        if self.write_count % self.auto_flush_every == 0:
            self.csv_file.flush()

    def close(self):
        try:
            self.csv_file.flush()
            self.csv_file.close()
        except Exception:
            pass

    def extract_and_save(
        self,
        session_id,
        video_id,
        ts_sec,
        frame_idx,
        person_id_raw,
        hand_assigner,
        kpts,
        scores,
        boxes,
    ):
        # 脖子原点 + 肩宽标尺
        if scores[5] < 0.2 or scores[6] < 0.2:
            return

        neck_x = (kpts[5][0] + kpts[6][0]) / 2.0
        neck_y = (kpts[5][1] + kpts[6][1]) / 2.0
        shoulder_width = math.hypot(kpts[5][0] - kpts[6][0], kpts[5][1] - kpts[6][1])
        shoulder_width = max(shoulder_width, 10.0)

        for hand_side, wrist_idx in [("left", 9), ("right", 10)]:
            score = float(scores[wrist_idx])
            if score < 0.15:
                continue

            wrist_x, wrist_y = float(kpts[wrist_idx][0]), float(kpts[wrist_idx][1])

            # A. 归一化位置
            norm_x = (wrist_x - neck_x) / shoulder_width
            norm_y = (wrist_y - neck_y) / shoulder_width

            # B. 稳定轨迹 ID（替代仅用 p_idx）
            track_id = hand_assigner.assign(hand_side, norm_x, norm_y, ts_sec)
            hand_key = f"{session_id}_{track_id}_{hand_side}"

            if hand_key not in self.history:
                self.history[hand_key] = deque(maxlen=3)

            history = self.history[hand_key]
            history.append((norm_x, norm_y, ts_sec))

            # C. 按 dt 计算速度/加速度，避免跳帧影响
            v_x, v_y, a_x, a_y = 0.0, 0.0, 0.0, 0.0
            if len(history) >= 2:
                x1, y1, t1 = history[-2]
                x2, y2, t2 = history[-1]
                dt = max(t2 - t1, 1e-3)
                v_x = (x2 - x1) / dt
                v_y = (y2 - y1) / dt

            if len(history) >= 3:
                x0, y0, t0 = history[-3]
                x1, y1, t1 = history[-2]
                x2, y2, t2 = history[-1]
                dt1 = max(t1 - t0, 1e-3)
                dt2 = max(t2 - t1, 1e-3)
                v1_x, v1_y = (x1 - x0) / dt1, (y1 - y0) / dt1
                v2_x, v2_y = (x2 - x1) / dt2, (y2 - y1) / dt2
                a_dt = max(t2 - t0, 1e-3)
                a_x = (v2_x - v1_x) / a_dt
                a_y = (v2_y - v1_y) / a_dt

            # D. 环境关系特征（沿用原逻辑）
            is_in_box = 0
            target_box_id = -1
            min_dist = float('inf')

            for box in boxes:
                contour = box['orig_contour']
                M_moments = cv2.moments(contour)
                if M_moments["m00"] != 0:
                    cx = float(M_moments["m10"] / M_moments["m00"])
                    cy = float(M_moments["m01"] / M_moments["m00"])
                else:
                    cx, cy = float(contour[0][0][0]), float(contour[0][0][1])

                dist_to_center = math.hypot(wrist_x - cx, wrist_y - cy) / shoulder_width
                if cv2.pointPolygonTest(contour, (wrist_x, wrist_y), False) >= 0:
                    is_in_box = 1
                    target_box_id = box['box_id']
                    min_dist = dist_to_center
                    break

                if dist_to_center < min_dist:
                    min_dist = dist_to_center
                    target_box_id = box['box_id']

            row = [
                session_id,
                video_id,
                round(ts_sec, 4),
                frame_idx,
                track_id,
                person_id_raw,
                hand_side,
                round(norm_x, 4),
                round(norm_y, 4),
                round(score, 4),
                round(v_x, 4),
                round(v_y, 4),
                round(a_x, 4),
                round(a_y, 4),
                is_in_box,
                round(min_dist, 4),
                target_box_id,
            ]
            self._save_row(row)
